// Add this CSS to get the blinking cursor
#typed-cursor{
  opacity: 1;
  font-weight: 100;
  // add prefixes
  animation: blink 0.7s infinite;
}

@-keyframes blink{
  0% { opacity:1; }
  50% { opacity:0; }
  100% { opacity:1; }
}

      $(".element").typed({
